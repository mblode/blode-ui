## Usage

```tsx
import { BarList } from "@/components/ui/bar-list";
```

```tsx
const data = [
  { name: "Search", value: 72 },
  { name: "Email", value: 58 },
  { name: "Social", value: 33 },
];

<BarList data={data} valueFormatter={(value) => `${value}%`} />;
```

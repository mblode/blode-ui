## Usage

```tsx
import {
  Meter,
  MeterIndicator,
  MeterLabel,
  MeterTrack,
  MeterValue,
} from "@/components/ui/meter";
```

```tsx
<Meter value={24}>
  <div className="flex items-center justify-between">
    <MeterLabel>Storage Used</MeterLabel>
    <MeterValue>{(formattedValue) => `${formattedValue}%`}</MeterValue>
  </div>
  <MeterTrack>
    <MeterIndicator />
  </MeterTrack>
</Meter>
```
